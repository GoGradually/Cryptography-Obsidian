1. 다항식의 거듭제곱 코드(빠른 거듭제곱 이용)
~~~cpp
uint16_t gf16_pow(uint16_t a, uint16_t b) {
	uint16_t ret = 1;
	while (b > 0) {
		if (b & 1) {
			ret = gf16_mul(ret, a);
		}
		a = gf16_mul(a, a);
		b >>= 1;
	}
	return ret;
}
~~~
